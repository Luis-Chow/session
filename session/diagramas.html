2 · Casos de Uso

3 · SQL del subsistema de negocio

El DDL en tu mismo estilo (snake_case, tabla_id SERIAL, tabla_na). Listo para pegar en sentences.json.

Ver script CREATE TABLE 
CREATE TABLE IF NOT EXISTS status (
  status_id SERIAL PRIMARY KEY,
  status_na VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS position (
  position_id SERIAL PRIMARY KEY,
  position_na VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS person (
  person_id SERIAL PRIMARY KEY,
  person_na VARCHAR(100) NOT NULL,
  person_ln VARCHAR(100),
  person_email VARCHAR(120),
  position_id INT REFERENCES position(position_id)
);

CREATE TABLE IF NOT EXISTS project (
  project_id SERIAL PRIMARY KEY,
  project_na VARCHAR(100) NOT NULL,
  project_de VARCHAR(255),
  start_date DATE,
  end_date DATE,
  status_id INT REFERENCES status(status_id)
);

CREATE TABLE IF NOT EXISTS project_role (
  project_role_id SERIAL PRIMARY KEY,
  project_role_na VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS project_role_person (
  prp_id SERIAL PRIMARY KEY,
  project_id      INT NOT NULL REFERENCES project(project_id),
  person_id       INT NOT NULL REFERENCES person(person_id),
  project_role_id INT NOT NULL REFERENCES project_role(project_role_id),
  UNIQUE (project_id, person_id, project_role_id)
);

CREATE TABLE IF NOT EXISTS activity (
  activity_id SERIAL PRIMARY KEY,
  activity_na VARCHAR(100) NOT NULL,
  activity_de VARCHAR(255),
  prp_id      INT NOT NULL REFERENCES project_role_person(prp_id),
  status_id   INT REFERENCES status(status_id)
);

CREATE TABLE IF NOT EXISTS notification (
  notification_id SERIAL PRIMARY KEY,
  activity_id  INT NOT NULL REFERENCES activity(activity_id),
  notify_date  DATE NOT NULL,
  notify_time  TIME NOT NULL,
  progress_pct NUMERIC(5,2) NOT NULL,
  observation  VARCHAR(255),
  hours_spent  NUMERIC(5,2)
);

CREATE TABLE IF NOT EXISTS audit (
  audit_id    SERIAL PRIMARY KEY,
  user_id     INT REFERENCES users(user_id),
  audit_date  DATE NOT NULL,
  audit_time  TIME NOT NULL,
  action      VARCHAR(20) NOT NULL,
  description VARCHAR(255),
  ip          VARCHAR(45)
);
